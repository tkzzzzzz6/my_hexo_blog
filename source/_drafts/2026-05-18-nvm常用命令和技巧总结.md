# 常用命令

```bash
# 安装nvm
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.1/install.sh | bash

# 安装node
nvm install node

# 切换node版本
nvm use node

# 查看已安装的node版本
nvm ls

# 查看可安装的node版本
nvm ls-remote

# 卸载node版本
nvm uninstall node

# 设置默认node版本
nvm alias default node

# 切换到特定版本
nvm use 14.17.0

# 切换到最新的LTS版本
nvm use --lts

# 切换到最新的稳定版本
nvm use stable

# 切换到上一个版本
nvm use --lts
```
